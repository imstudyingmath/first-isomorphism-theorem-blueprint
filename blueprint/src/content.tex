\begin{theorem}[First Isomorphism Theorem]
  \lean{my_first_iso} 
  \leanok
  Let $G$ and $H$ be groups, and let $f: G \to H$ be a group homomorphism.
  Then, the quotient group $G/\ker f$ is isomorphic to the image of $f$, denoted by $\mathrm{im} f$.
\end{theorem}

\begin{proof}
  \leanok
  We will prove this using QuotientGroup.quotientKerEquivRange from Lean's standard library (Mathlib).
\end{proof}
